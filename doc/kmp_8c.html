step 1 -> ! bbcabcdababcdabcdabde abcdabd

step 1 -> ! bbcabcdababcdabcdabde abcdabd

step 1 -> ! bbcabcdababcdabcdabde abcdabd

step 1 -> ... until find failure charactor 'a'-'d' ! * bbcabcdababcdabcdabde abcdabd

step next[5] = 4, then step 1... until find failure charactor 'a'-'c' ! * bbcabcdababcdabcdabde abcdabd

step next[3] = 2 ... ! bbcabcdababcdabcdabde abcdabd
